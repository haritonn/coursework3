def pairwise_metrics(y_true, y_pred):
    tn, fp, fn, tp = pair_confusion_matrix(y_true, y_pred).ravel()

    precision = tp / (tp + fp) if tp + fp > 0 else 0
    recall = tp / (tp + fn) if tp + fn > 0 else 0

    f1 = 2 * precision * recall / (precision + recall) if precision + recall > 0 else 0

    return precision, recall, f1

def compute_true_paths(df, col="hF1_ancestors", sep="|"):
    return [
        [node for node in value.split(sep) if node]
        for value in df[col].astype(str)
    ]


def common_prefix(paths):
    if len(paths) == 0:
        return []

    prefix = []
    for nodes in zip(*paths):
        if len(set(nodes)) != 1:
            break
        prefix.append(nodes[0])

    return prefix


def compute_pred_paths(y_pred, true_paths, noise_label=-1, root="root"):
    y_pred = np.asarray(y_pred)
    label_to_path = {}

    for label in set(y_pred):
        if label == noise_label:
            label_to_path[label] = [root]
        else:
            idx = np.where(y_pred == label)[0]
            label_to_path[label] = common_prefix([true_paths[i] for i in idx])

    return [label_to_path[label] for label in y_pred]


def compute_hf1(true_paths, pred_paths):
    intersection_sum = 0
    true_sum = 0
    pred_sum = 0

    for pred_path, true_path in zip(pred_paths, true_paths):
        pred_set = set(pred_path)
        true_set = set(true_path)

        intersection_sum += len(pred_set & true_set)
        true_sum += len(true_set)
        pred_sum += len(pred_set)

    h_precision = intersection_sum / pred_sum if pred_sum else 0
    h_recall = intersection_sum / true_sum if true_sum else 0
    h_f1 = 2 * h_precision * h_recall / (h_precision + h_recall) if h_precision + h_recall else 0

    return h_precision, h_recall, h_f1

def tfidf_preprocessing(text):
    text = re.sub(r"[^а-яё\s]", " ", text.lower())
    result = []

    for token in text.split():
        if len(token) < 2: continue
        lemma = MORPH.parse(token)[0].normal_form
        if lemma in RU_STOPWORDS: continue

        result.append(lemma)

    return " ".join(result)

def clustering_metrics(X, y_true, y_pred, true_paths=None, pred_paths=None):
    precision, recall, f1 = pairwise_metrics(y_true, y_pred)
    ari = adjusted_rand_score(y_true, y_pred)

    if true_paths is not None and pred_paths is not None:
        h_precision, h_recall, h_f1 = compute_hf1(true_paths, pred_paths)
    else:
        h_precision, h_recall, h_f1 = np.nan, np.nan, np.nan

    mask = y_pred != -1
    internal = {
        "silhouette": np.nan,
        "calinski_harabasz": np.nan,
        "davies_bouldin": np.nan,
    }

    if len(set(y_pred[mask])) >= 2:
        internal["silhouette"] = silhouette_score(X[mask], y_pred[mask])
        internal["calinski_harabasz"] = calinski_harabasz_score(X[mask], y_pred[mask])
        internal["davies_bouldin"] = davies_bouldin_score(X[mask], y_pred[mask])

    return {
        "precision": precision,
        "recall": recall,
        "f1-score": f1,
        "h_precision": h_precision,
        "h_recall": h_recall,
        "h_f1-score": h_f1,
        "ari": ari,
        **internal
    }

def reduce_embeddings(X, pca_dim=100, umap_dim=15):
    X = normalize(X)
    pca_dim = min(pca_dim, X.shape[1], X.shape[0] - 1)
    X_pca = PCA(
        n_components = pca_dim,
        random_state=42
    ).fit_transform(X)

    X_umap = UMAP(
        n_components=umap_dim,
        metric="cosine",
        random_state=42
    ).fit_transform(X_pca)

    return X_umap

def run_clusterers(X, n_clusters, min_cluster_size=5):
    kmeans = KMeans(
        n_clusters=n_clusters,
        random_state=42,
        n_init="auto"
    ).fit(X)

    hdbscan = HDBSCAN(
        min_cluster_size=min_cluster_size,
        metric="euclidean"
    ).fit(X)

    agglomerative = AgglomerativeClustering(
        n_clusters=n_clusters,
        linkage="ward"
    ).fit(X)

    return {
        "kmeans": kmeans.labels_,
        "hdbscan": hdbscan.labels_,
        "agglomerative": agglomerative.labels_,
    }

def full_pipeline(embeddings_dict, n_clusters=71, min_cluster_sizes=[5, 10, 15, 20]):
    results_total = {}
    for idx, min_cluster_size in enumerate(min_cluster_sizes):
        results_clustering = []
        print(f"{idx+1} out of {len(min_cluster_sizes)}")
        for emb_name, X_raw in embeddings_dict.items():
            X_reduced = reduce_embeddings(
                X_raw,
                pca_dim=100,
                umap_dim=15
            )
        
            clusterers_output = run_clusterers(
                X_reduced,
                n_clusters=n_clusters,
                min_cluster_size=min_cluster_size
            )
        
            for algo_name, y_pred in clusterers_output.items():
                pred_paths = compute_pred_paths(y_pred, true_paths)
                metrics = clustering_metrics(
                    X=X_reduced,
                    y_true=y_true,
                    y_pred=y_pred,
                    true_paths=true_paths,
                    pred_paths=pred_paths,
                )
        
                results_clustering.append({
                    "embeddings": emb_name,
                    "algorithm": algo_name,
                    **metrics,
                    "n_pred_clusters": len(set(y_pred)) - int(-1 in y_pred),
                    "n_noise": int(np.sum(y_pred == -1))
                })
        
        results_df = pd.DataFrame(results_clustering)
        results_df = results_df[
            [
                "algorithm",
                "embeddings",
                "precision",
                "recall",
                "f1-score",
                "h_precision",
                "h_recall",
                "h_f1-score",
                "ari",
                "silhouette",
                "calinski_harabasz",
                "davies_bouldin",
                "n_pred_clusters",
                "n_noise",
            ]
        ]
        results_total[f"res_{min_cluster_size}"] = results_df
    
    return results_total

def plot_hdbscan_results(savename=None):
    cols_to_plot = [
        "precision", "recall", "f1-score", "h_f1-score", "ari",
        "silhouette", "calinski_harabasz", "davies_bouldin"
    ]

    plot_data = total_df[total_df["algorithm"] == "hdbscan"].copy()

    embeddings = plot_data["embeddings"].unique()
    colors = dict(zip(embeddings, plt.get_cmap("tab10").colors))

    n_cols = 4
    n_rows = int(np.ceil(len(cols_to_plot) / n_cols))
    fig, axes = plt.subplots(nrows=n_rows, ncols=n_cols, figsize=(20, 4.5 * n_rows))
    axes = axes.flatten()

    for metric, ax in zip(cols_to_plot, axes):
        for emb_name in embeddings:
            filtered_data = plot_data[plot_data["embeddings"] == emb_name].sort_values("min_cluster_size")

            ax.plot(
                filtered_data["min_cluster_size"],
                filtered_data[metric],
                marker="o",
                linewidth=2,
                markersize=4,
                color=colors[emb_name],
                label=emb_name
            )

            if metric == "davies_bouldin": best_idx = filtered_data[metric].idxmin()
            else: best_idx = filtered_data[metric].idxmax()

            best_row = filtered_data.loc[best_idx]
            best_mcs = best_row["min_cluster_size"]
            best_value = best_row[metric]

            ax.axvline(best_mcs, color=colors[emb_name], linestyle="--", alpha=0.25)

            ax.text(
                best_mcs,
                best_value,
                f"{int(best_mcs)}",
                fontsize=8,
                ha="center",
                va="bottom",
                color=colors[emb_name]
            )

        ax.set_title(metric)
        ax.set_xlabel("min_cluster_size")
        ax.grid(alpha=0.3)
        ax.spines["top"].set_visible(False)
        ax.spines["right"].set_visible(False)

    for ax in axes[len(cols_to_plot):]:
        ax.set_visible(False)

    handles, labels = axes[0].get_legend_handles_labels()
    by_label = dict(zip(labels, handles))

    fig.legend(by_label.values(), by_label.keys(), loc="lower center", ncols=len(by_label), frameon=False)

    fig.suptitle("HDBSCAN results by min_cluster_size", fontsize=16)
    plt.tight_layout(rect=[0, 0.08, 1, 0.95])
    plt.show()

    if savename is not None:
        fig.savefig(f"{savename}.jpg")
